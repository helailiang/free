实时 TCP 点云接收器

支持：

HTTP 获取 handle
HTTP 启动扫描
HTTP watchdog
TCP 持续接收
自动切帧
自动拼完整 scan
latest_scan 实时覆盖
实时获取最新点云

```python
import socket
import struct
import threading
import time
from dataclasses import dataclass
from typing import Optional

import numpy as np
import requests


MAGIC = 0xA25C
PACKET_TYPE = 0x0043


# =========================
# 数据结构
# =========================

@dataclass
class ScanFrame:
    scan_count: int
    timestamp: int

    angles: np.ndarray
    distances: np.ndarray
    amplitudes: np.ndarray


# =========================
# 雷达客户端
# =========================

class LidarClient:

    def __init__(self, sensor_ip: str):

        self.sensor_ip = sensor_ip

        self.handle = None
        self.port = None

        self.sock = None

        self.running = False

        self.recv_thread_obj = None
        self.watchdog_thread_obj = None

        self.buffer = bytearray()

        self.latest_scan: Optional[ScanFrame] = None

        self.scan_lock = threading.Lock()

        # 当前正在拼接的 scan
        self.scan_cache = {}

    # ==========================================
    # HTTP
    # ==========================================

    def request_handle(self):

        url = f"http://{self.sensor_ip}/cmd/request_handle_tcp"

        r = requests.get(url, timeout=2)

        r.raise_for_status()

        data = r.json()

        if data["error_code"] != 0:
            raise RuntimeError(data)

        self.handle = data["handle"]
        self.port = data["port"]

        print(f"[INFO] handle={self.handle}")
        print(f"[INFO] tcp_port={self.port}")

    def start_scan_output(self):

        url = (
            f"http://{self.sensor_ip}/cmd/start_scanoutput"
            f"?handle={self.handle}"
        )

        r = requests.get(url, timeout=2)

        r.raise_for_status()

        data = r.json()

        if data["error_code"] != 0:
            raise RuntimeError(data)

        print("[INFO] scan output started")

    def feed_watchdog(self):

        url = (
            f"http://{self.sensor_ip}/cmd/feed_watchdog"
            f"?handle={self.handle}"
        )

        try:
            requests.get(url, timeout=1)
        except Exception as e:
            print("[WARN] watchdog:", e)

    # ==========================================
    # TCP
    # ==========================================

    def connect_tcp(self):

        self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

        self.sock.setsockopt(socket.IPPROTO_TCP, socket.TCP_NODELAY, 1)

        self.sock.settimeout(2)

        self.sock.connect((self.sensor_ip, self.port))

        print("[INFO] TCP connected")

    # ==========================================
    # 启动
    # ==========================================

    def start(self):

        self.request_handle()

        self.start_scan_output()

        self.connect_tcp()

        self.running = True

        self.recv_thread_obj = threading.Thread(
            target=self.recv_loop,
            daemon=True,
        )

        self.watchdog_thread_obj = threading.Thread(
            target=self.watchdog_loop,
            daemon=True,
        )

        self.recv_thread_obj.start()
        self.watchdog_thread_obj.start()

    def stop(self):

        self.running = False

        if self.sock:
            self.sock.close()

    # ==========================================
    # watchdog
    # ==========================================

    def watchdog_loop(self):

        while self.running:

            self.feed_watchdog()

            time.sleep(1)

    # ==========================================
    # TCP 接收
    # ==========================================

    def recv_loop(self):

        while self.running:

            try:

                data = self.sock.recv(65535)

                if not data:
                    print("[ERROR] tcp disconnected")
                    break

                self.buffer.extend(data)

                while True:

                    packet = self.extract_packet()

                    if packet is None:
                        break

                    self.handle_packet(packet)

            except socket.timeout:
                continue

            except Exception as e:
                print("[ERROR] recv_loop:", e)
                break

    # ==========================================
    # TCP 切帧
    # ==========================================

    def extract_packet(self):

        while True:

            if len(self.buffer) < 8:
                return None

            magic = struct.unpack_from("<H", self.buffer, 0)[0]

            if magic != MAGIC:
                self.buffer.pop(0)
                continue

            packet_len = struct.unpack_from("<I", self.buffer, 4)[0]

            if packet_len < 32:
                self.buffer.pop(0)
                continue

            if len(self.buffer) < packet_len:
                return None

            packet = bytes(self.buffer[:packet_len])

            del self.buffer[:packet_len]

            return packet

    # ==========================================
    # packet 解析
    # ==========================================

    def handle_packet(self, packet: bytes):

        try:

            # -----------------------------------
            # 基础头
            # -----------------------------------

            packet_type = struct.unpack_from("<H", packet, 2)[0]

            if packet_type != PACKET_TYPE:
                return

            packet_len = struct.unpack_from("<I", packet, 4)[0]

            header_len = struct.unpack_from("<H", packet, 8)[0]

            scan_count = struct.unpack_from("<H", packet, 10)[0]

            packet_no = struct.unpack_from("<H", packet, 12)[0]

            timestamp = struct.unpack_from("<Q", packet, 14)[0]

            status_flags = struct.unpack_from("<I", packet, 30)[0]

            scan_frequency = struct.unpack_from("<I", packet, 34)[0]

            points_per_scan = struct.unpack_from("<H", packet, 38)[0]

            point_count = struct.unpack_from("<H", packet, 40)[0]

            first_point_index = struct.unpack_from("<H", packet, 42)[0]

            first_angle_raw = struct.unpack_from("<i", packet, 44)[0]

            angle_step_raw = struct.unpack_from("<i", packet, 48)[0]

            first_angle = first_angle_raw / 10000.0
            angle_step = angle_step_raw / 10000.0

            # -----------------------------------
            # 初始化 scan cache
            # -----------------------------------

            if scan_count not in self.scan_cache:

                self.scan_cache[scan_count] = {
                    "distances": np.full(
                        points_per_scan,
                        -1,
                        dtype=np.int32,
                    ),

                    "amplitudes": np.zeros(
                        points_per_scan,
                        dtype=np.uint16,
                    ),

                    "angles": np.zeros(
                        points_per_scan,
                        dtype=np.float32,
                    ),

                    "filled": np.zeros(
                        points_per_scan,
                        dtype=np.bool_,
                    ),

                    "timestamp": timestamp,
                }

            cache = self.scan_cache[scan_count]

            # -----------------------------------
            # 点数据解析
            # -----------------------------------

            offset = header_len

            for i in range(point_count):

                v = struct.unpack_from("<I", packet, offset)[0]

                offset += 4

                distance = v & 0xFFFFF
                amplitude = (v >> 20) & 0xFFF

                idx = first_point_index + i

                if idx >= points_per_scan:
                    continue

                if distance == 0xFFFFF:
                    distance = -1

                angle = first_angle + i * angle_step

                cache["distances"][idx] = distance
                cache["amplitudes"][idx] = amplitude
                cache["angles"][idx] = angle

                cache["filled"][idx] = True

            # -----------------------------------
            # 一整圈完成
            # -----------------------------------

            if np.all(cache["filled"]):

                frame = ScanFrame(
                    scan_count=scan_count,
                    timestamp=cache["timestamp"],
                    angles=cache["angles"].copy(),
                    distances=cache["distances"].copy(),
                    amplitudes=cache["amplitudes"].copy(),
                )

                with self.scan_lock:
                    self.latest_scan = frame

                # 清理旧 scan
                del self.scan_cache[scan_count]

                # 防止 cache 无限增长
                if len(self.scan_cache) > 10:

                    oldest = sorted(self.scan_cache.keys())[0]

                    del self.scan_cache[oldest]

        except Exception as e:

            print("[ERROR] handle_packet:", e)

    # ==========================================
    # 获取最新 scan
    # ==========================================

    def get_latest_scan(self) -> Optional[ScanFrame]:

        with self.scan_lock:
            return self.latest_scan


# ==============================================
# DEMO
# ==============================================

if __name__ == "__main__":

    lidar = LidarClient("192.168.1.50")

    lidar.start()

    while True:

        scan = lidar.get_latest_scan()

        if scan is not None:

            valid_count = np.sum(scan.distances > 0)

            print(
                f"scan={scan.scan_count} "
                f"points={valid_count}"
            )

        time.sleep(0.1)
```

1. 实时 latest_scan

永远最新。

不会积压。

2. TCP 自动切帧

支持：

半包
粘包
多包
3. scan 自动拼接

scan_count
first_point_index
